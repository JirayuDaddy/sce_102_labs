#include <stdio.h>
#include <math.h>

int main()
{
    float radius, area , pi = 3.1416;
    printf("Enter r : ");
    scanf("%f",&radius);

    area = pi * radius * radius;

    printf("Circle area = %.2f",area);

    condition = (area >=0 ) && (area >= 100);
    printf("Big circle %d",condition);
    return 0;
}
